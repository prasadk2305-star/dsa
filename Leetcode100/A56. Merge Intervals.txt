/*
   here we used two linked list function that is getlast these function help you to ftech value as well as update value 
*/
class Solution {
    public int[][] merge(int[][] intervals) {
      Arrays.sort(intervals,(a,b)->Integer.compare(a[0],b[0]));   // sort only on the basis of starting index
      LinkedList<int []> ans = new LinkedList<>();
      for(int[] i : intervals)
      {
           if(ans.isEmpty() || ans.getLast()[1]<i[0])    // if array is empty then we need to add the elemnt directly 
           {
               ans.add(i);
           }else{
              ans.getLast()[1] = Math.max(ans.getLast()[1],i[1]);
           }
      }  
      return ans.toArray(new int[ans.size()][]);   // also at the time of returning we convert linkedlist into array by craeting new array of linkedlist size
    }
}
