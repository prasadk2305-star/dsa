class Solution {
    public int lastStoneWeight(int[] stones) {
        PriorityQueue<Integer> q = new PriorityQueue<>(Collections.reverseOrder());
        for(int i:stones)
        {
            q.add(i);
        }
        while(!q.isEmpty() && q.size()>=2)
        {
             int a = q.poll();
             int b = q.poll();
             if(a!=b)
             {
                q.offer(a-b);
             }
        }
        return q.size()==1?q.poll():0;

    }
}
