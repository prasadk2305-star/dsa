/*

rob1  = is always the value of max upto last step;
rob2 = is always the value max upto current step

*/

class Solution {
    public int rob(int[] nums) {
        int rob1 = 0;
        int rob2 = 0;
        int max = 0;
        for(int i=0; i<nums.length; i++)
        {
            max = Math.max(nums[i]+rob1,max);
            rob1 = rob2;
            rob2 = max;
        }
        return max;
    }
}
