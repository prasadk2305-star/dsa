
/*



sliding windows core logic is just focus on one side means mostly right 
as per condition iterate through right
and whenever condition fail moves left 


*/
class Solution {
    public int lengthOfLongestSubstring(String s) {
        int l=0,r=0;
        int max= 0;
        HashSet<Character> set = new HashSet<>();
        while(r<s.length()){
               if(!set.contains(s.charAt(r)))
               {
                   set.add(s.charAt(r));
                   r++;
                   max = Math.max(max,r-l);
               }else{
                   set.remove(s.charAt(l));
                   l++;
               }
              
        }
        return max;
    }
}
