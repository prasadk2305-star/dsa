/*
    here logic is to track the last index of each chracter
      and on the basis of that tarck the start and end index of the charcter where end is the max index of charcter in that parttion upto that our parttion is valid
      if i == end at that time we just count partitoin length and add it into the ans and set start and end as i+1
*/

class Solution {
    public List<Integer> partitionLabels(String s) {
        int[] a = new int[26];
        for(int i=0; i<s.length(); i++)
        {
            a[s.charAt(i)-'a'] = i;
        }
        ArrayList<Integer> ans = new ArrayList<>();
        int curr = 0;
        int start = 0;
        for(int i=0; i<s.length(); i++)
        {
           curr = Math.max(curr,a[s.charAt(i)-'a']);
           if(curr == i){
              ans.add(curr-start+1);
              start = i+1;
              curr = i+1;
           }
        }
        return ans;
    }
}
