class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
       ListNode dummy = new ListNode(1);
       ListNode ans = dummy;
       int carry = 0;
       while(l1!=null || l2!=null)
       {

            int a = l1==null?0:l1.val;
            int b = l2==null?0:l2.val;
            int c = a+b+carry;      //  must do total as at calculation doing total may leads to further trouble
            carry = c/10;
            int val = c%10;
            ListNode curr = new ListNode(val);    // and as per linkedlist we didnt add val directly so for that we need to create first Tempeorary node
            ans.next = curr;
            if(l1!=null) l1 = l1.next;
            if(l2!=null) l2 = l2.next;
            ans = ans.next;
       } 
       if(carry!=0)
       {
            ListNode curr = new ListNode(carry);
            ans.next = curr;
       }
       return dummy.next;
    }
}
