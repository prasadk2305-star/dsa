/*
  core logic is think current 
  because if any previous entry causes harm to logic that we are handling by calculating totalgas and totalcost 
  if totalcost is greter than totalgas then it an obvious that past logic fail and we return -1 otherwise we gona return current

*/

class Solution {
    public int canCompleteCircuit(int[] gas, int[] cost) {
        int currsum=0;
        int curr=0;
        int totalgas = 0;
        int totalcost= 0;
        for(int i=0; i<gas.length; i++)
        {
            totalgas += gas[i];
            totalcost += cost[i];
            currsum += gas[i] - cost[i];
            if(currsum<0){
                currsum = 0;
                curr = i+1;
            }
        }
        if(totalgas<totalcost)
        {
            return -1;
        }
        return curr;
    }
}
