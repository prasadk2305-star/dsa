class Solution {
    public int getSecondLargest(int[] arr) {
        // code here
      int max = arr[0];
      for(int i=1; i<arr.length; i++)
      {
           if(arr[i]>max)
           {
               max = arr[i];
           }
      }
      int sec = Integer.MIN_VALUE;
      for(int i=0; i<arr.length; i++)
      {
          if(arr[i]<max && arr[i]>sec)
          {
              sec = arr[i];
          }
      }
      return sec;
    }
}
