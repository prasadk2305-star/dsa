/*
here logic is simple that make two new linkedlist 
one for odd idex nodes and even index nodes
upadte everyone next with its second next
and at the end odd.next = even ll 's head we add


*/
class Solution {
    public ListNode oddEvenList(ListNode head) {
        if(head==null || head.next==null) return head;
        ListNode dummyodd = new ListNode(0);
        ListNode dummyeven = new ListNode(0);
        ListNode odd  = head;
        dummyodd.next = odd;
        ListNode even = head.next;
        dummyeven.next = even;
       
        while(even!=null && even.next!=null)   // here we check only even beacuse in all possible cases even node comes first at the end
        {
            odd.next = odd.next.next;
            even.next = even.next.next;
            odd = odd.next;
            even = even.next;
        }
        odd.next = dummyeven.next;
        return dummyodd.next;
       
    }
}
