/*
here also we did same trick as we did in remove kth node from end 

we consider dummy node as starring point of slow
that help us to keep the slow one step before the actual mid and as always we return dummy.next

but here the fast is unchangead as his works is need as always 

*/
class Solution {
    public ListNode deleteMiddle(ListNode head) {
     ListNode dummy = new ListNode(0);
     dummy.next = head;
     ListNode slow = dummy;
     ListNode fast = head;
     while(fast!=null && fast.next!=null)
     {
         slow = slow.next;
         fast = fast.next.next;
     }   
     slow.next = slow.next.next;
     return dummy.next;
    }
}
