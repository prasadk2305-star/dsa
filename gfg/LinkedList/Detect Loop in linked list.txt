/*
   check loop is present in linked list or not is the question

*/
class Solution {
    public boolean detectLoop(Node head) {
        // code here
        Node slow = head;
        Node fast = head;
        while(fast!=null  && fast.next!=null)
        {
            slow = slow.next;
            fast = fast.next.next;
            if(fast == slow) return true;
        }
        
        return false;
    }
}
