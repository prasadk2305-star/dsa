class Solution {
    public boolean isSorted(int[] arr) {
        // code here
        if(arr.length == 1)
        {
            return true;
        }
        int count = 0;
        if(arr[0]<arr[arr.length-1]){
            count++;
        }
        for(int i=1; i<arr.length; i++){
            if(arr[i]<arr[i-1]){
                count++;                
            }
        }
        
        return count==1?true:false;
    }
}
