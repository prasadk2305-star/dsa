class Solution {
    public List<Integer> generatebyrow(int n)
    {
         List<Integer> ans = new ArrayList<>();
         ans.add(1);
         for(int i=1; i<n; i++)
         {
              int r = n-1;
              int curr = 1;
              for(int j=0; j<i; j++)
              {
                  curr = (curr * (r-j)) / (j+1);
              }
              ans.add(curr);
         }
         return ans;
    }
    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> final1 = new ArrayList<>();
        for(int i=1; i<=numRows; i++)
        {
            final1.add(generatebyrow(i));
        }
        return final1;
    }
}

       List<Integer> ans = new ArrayList<>();

        for(int i=1; i<rowIndex+1; i++)
        {
             int r = rowIndex;
             long curr =1;
             for(int j=0; j<i; j++)

                   curr = curr * (r-j) / (j+1);
