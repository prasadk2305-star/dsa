here some minor chnages are there like the floor in array problem 
  the main changes is that we have given same number that time we dont neeed to passed lowest index we have to return that number 
  so here logic is evolved around finding the correct number not index
  also here in ceil and floor we are stroing direct number.

*/

class Solution {
    public int[] getFloorAndCeil(int x, int[] arr) {
        Arrays.sort(arr);
        int floor = -1;
        int ceil = -1;
        int left = 0;
        int right = arr.length-1;
        
        while(left<=right)
        {
            int mid = left+(right-left)/2;
            if(arr[mid]>x)
            {
                ceil = arr[mid];
                right = mid-1;
            }else if(arr[mid] ==x ){
                ceil = arr[mid];
                floor = mid;
                break;
                
            }else{
                floor = mid;
                left = mid+1;
            }
        }
      
        
        return new int[]{floor,ceil};
        
        
    }
}
