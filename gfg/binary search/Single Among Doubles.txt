class Sol {
    public static int search(int n, int arr[]) {
        // your code here
        int left = 0;
        int right = arr.length-1;
        while(left<right)
        {
          int mid = left+(right-left)/2;
          if(arr[mid] == arr[mid+1])
          {
              mid = mid+1;
          }
          int len = mid-left+1;
          if(len%2==0)
          {
              left = mid+1;
          }else{
              right= mid-1;
          }

        }
      return arr[left];
    }
}
